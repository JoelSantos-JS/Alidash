rules_version = '2';

service cloud.firestore {
  match /databases/{database}/documents {
    
    // Regras para a coleção users (se existir)
    match /users/{userId} {
      allow read, create: if request.auth != null;
      allow update: if request.auth != null && request.auth.uid == userId;
    }
    
    // Regras para a coleção user-data (PRINCIPAL)
    match /user-data/{userId} {
      // Usuários podem acessar apenas seus próprios dados
      allow read, write: if request.auth != null && request.auth.uid == userId;
      
      // Admin específico pode ler qualquer documento (para backup)
      allow read: if request.auth != null && request.auth.uid == 'joeltere9@gmail.com';
    }
    
    // Negar acesso a outras coleções por padrão
    match /{document=**} {
      allow read, write: if false;
    }
  }
}
